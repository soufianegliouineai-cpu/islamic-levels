شغّل في console المتصفح للتحقق من أن الإصلاحات تعمل بشكل صحيح.

اختبارات سريعة (افتح DevTools والصق في Console)

1. اختبار hashLocalPassword
// Should produce a base64 SHA-256 hash, not the old 'h_' format
const salt = generateLocalSalt();
const hash = await secureHashLocalPassword('test123', salt);
console.assert(!hash.startsWith('h_'), 'should not start with h_');
console.assert(hash.length > 20, 'should be long base64');
console.log('hash:', hash);
2. اختبار rolloverIfNewDay
// Should clear todayTasks, todayPrayers, todayAdhkar when day changes
state.todayTasks = [0, 1, 2];
state.lastDate = '2020-01-01';
rolloverIfNewDay();
console.assert(state.todayTasks.length === 0, 'tasks cleared');
console.assert(state.lastDate === getToday(), 'date updated');
3. اختبار clampCounters
state.gems = -100;
state.streak = -5;
state.totalDays = -10;
clampCounters();
console.assert(state.gems === 0, 'gems clamped to 0');
console.assert(state.streak === 0, 'streak clamped to 0');
console.assert(state.totalDays === 0, 'totalDays clamped to 0');
